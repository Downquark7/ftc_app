25   package org.firstinspires.ftc.teamcode; 
26    
27   import com.qualcomm.robotcore.hardware.HardwareMap; 
28   import com.qualcomm.robotcore.hardware.I2cAddr; 
29   import com.qualcomm.robotcore.hardware.I2cDevice; 
30   import com.qualcomm.robotcore.hardware.I2cDeviceSynch; 
31   import com.qualcomm.robotcore.hardware.I2cDeviceSynchImpl; 
32    
33   import java.util.Locale; 
34    
35   import static org.firstinspires.ftc.teamcode.robotconfig.dl; 
36    
37   /** 
38    * Created by Chris D on 10/5/2016. 
39    * <p> 
40    * Partially based on: 
41    * https://github.com/OliviliK/FTC_Library/blob/master/TCS34725_ColorSensor.java 
42    */ 
43   public class MultiplexColorSensor { 
44    
45       // Registers 
46       static final int ENABLE = 0x80; 
47       static final int ATIME = 0x81; 
48       static final int CONTROL = 0x8F; 
49       static final int ID = 0x92; 
50       static final int STATUS = 0x93; 
51       static final int CDATAL = 0x94; 
52    
53       // Default I2C address for multiplexer. The address can be changed to any 
54       // value from 0x70 to 0x77, so this line would need to be changed if a 
55       // non-default address is to be used. 
56       static final I2cAddr MUX_ADDRESS = new I2cAddr(0x70); 
57       // I2C address for color sensor 
58       static final I2cAddr ADA_ADDRESS = new I2cAddr(0x29); 
59       public static int GAIN_1X = 0x00; 
60       public static int GAIN_4X = 0x01; 
61       public static int GAIN_16X = 0x02; 
62       public static int GAIN_60X = 0x03; 
63       private I2cDevice mux; 
64       private I2cDeviceSynch muxReader; 
65       // Only one color sensor is needed in code as the multiplexer switches 
66       // between the physical sensors 
67       private byte[] adaCache; 
68       private I2cDevice ada; 
69       private I2cDeviceSynch adaReader; 
70       private int[] sensorPorts; 
71       private int lastPort; 
72    
73       /** 
74        * Initializes Adafruit color sensors on the specified ports of the I2C 
75        * multiplexer. 
76        * 
77        * @param hardwareMap  hardwareMap from OpMode 
78        * @param muxName      Configuration name of I2CDevice for multiplexer 
79        * @param colorName    Configuration name of I2CDevice for color sensor 
80        * @param ports        Out ports on multiplexer with color sensors attached 
81        * @param milliSeconds Integration time in milliseconds 
82        * @param gain         Gain (GAIN_1X, GAIN_4X, GAIN_16X, GAIN_60X) 
83        */ 
84       public MultiplexColorSensor(HardwareMap hardwareMap, 
85                                   String muxName, 
86                                   String colorName, 
87                                   int[] ports, 
88                                   double milliSeconds, 
89                                   int gain) { 
90           sensorPorts = ports; 
91    
92           mux = hardwareMap.i2cDevice.get(muxName); 
93           muxReader = new I2cDeviceSynchImpl(mux, MUX_ADDRESS, false); 
94           muxReader.engage(); 
95    
96           // Loop over the ports activating each color sensor 
97           for (int i = 0; i < sensorPorts.length; i++) { 
98               // Write to given output port on the multiplexer 
99               muxReader.write8(0x0, 1 << sensorPorts[i], true); 
100   
101              ada = hardwareMap.i2cDevice.get(colorName); 
102              adaReader = new I2cDeviceSynchImpl(ada, ADA_ADDRESS, false); 
103              adaReader.engage(); 
104   
105              final int time = integrationByte(milliSeconds); 
106              adaReader.write8(ENABLE, 0x03, true);  // Power on and enable ADC 
107              adaReader.read8(ID);                   // Read device ID 
108              adaReader.write8(CONTROL, gain, true); // Set gain 
109              adaReader.write8(ATIME, time, true);   // Set integration time 
110          } 
111      } 
112   
113      /** 
114       * Set the integration time on all the color sensors 
115       * 
116       * @param milliSeconds Time in millseconds 
117       */ 
118      public void setIntegrationTime(double milliSeconds) { 
119          int val = integrationByte(milliSeconds); 
120   
121          for (int i = 0; i < sensorPorts.length; i++) { 
122              muxReader.write8(0x0, 1 << sensorPorts[i], true); 
123              adaReader.write8(ATIME, val, true); 
124          } 
125      } 
126   
127      private int integrationByte(double milliSeconds) { 
128          int count = (int) (milliSeconds / 2.4); 
129          if (count < 1) count = 1;   // Clamp the time range 
130          if (count > 256) count = 256; 
131          return (256 - count); 
132      } 
133   
134      // Un-needed? 
135      public void startPolling() { 
136          for (int i = 0; i < sensorPorts.length; i++) { 
137              muxReader.write8(0x0, 1 << sensorPorts[i], true); 
138              adaReader.read8(STATUS); 
139          } 
140      } 
141   
142      /** 
143       * Retrieve the color read by the given color sensor 
144       * 
145       * @param port Port on multiplexer of given color sensor 
146       * @return Array containing the Clear, Red, Green, and Blue color values 
147       */ 
148      public int[] getCRGB(int port) { 
149          // Write to I2C port on the multiplexer 
150          if (port!= lastPort) {      // might speed things up a tad, as we only set the mux when changing between sensors 
151              muxReader.write8(0x0, 1 << port, true); 
152              lastPort = port;        // not on consecutive reads of the same sensor 
153          } 
154   
155          // Read color registers 
156          adaCache = adaReader.read(CDATAL, 8); 
157   
158          // Combine high and low bytes 
159          int[] crgb = new int[4]; 
160          for (int i = 0; i < 4; i++) { 
161              crgb[i] = (adaCache[2 * i] & 0xFF) + (adaCache[2 * i + 1] & 0xFF) * 256; 
162          } 
163   
164          robotconfig.addlog(dl, "read color sensor port: crgb", String.format(Locale.ENGLISH, "%d: %d %d %d %d", port, crgb[0], crgb[1], crgb[2], crgb[3])); 
165          return crgb; 
166      } 
167  }
