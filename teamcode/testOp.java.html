33   package org.firstinspires.ftc.teamcode; 
34    
35   import com.qualcomm.robotcore.eventloop.opmode.OpMode; 
36   import com.qualcomm.robotcore.eventloop.opmode.TeleOp; 
37    
38    
39   @TeleOp(name = "testOp", group = "2016") 
40    
41   public class testOp extends OpMode { 
42    
43       /* Declare OpMode members. */ 
44       robotconfig robot = new robotconfig(); 
45    
46       /* 
47        * Code to run ONCE when the driver hits INIT 
48        */ 
49       @Override 
50       public void init() { 
51           /* Initialize the hardware variables. 
52            * The init() method of the hardware class does all the work here 
53            */ 
54           robot.init(this); 
55           robot.move(0, 0, 0); 
56           // Send telemetry message to signify robot waiting; 
57           telemetry.addData("Say", "Hello Driver"); 
58           updateTelemetry(telemetry); 
59       } 
60    
61       /* 
62        * Code to run REPEATEDLY after the driver hits INIT, but before they hit PLAY 
63        */ 
64       @Override 
65       public void init_loop() { 
66       } 
67    
68       /* 
69        * Code to run ONCE when the driver hits PLAY 
70        */ 
71       @Override 
72       public void start() { 
73       } 
74    
75       /* 
76        * Code to run REPEATEDLY after the driver hits PLAY but before they hit STOP 
77        */ 
78       @Override 
79       public void loop() { 
80           double forward; 
81           double right; 
82           double spin; 
83    
84           //get values from joystick 
85           forward = -scale(gamepad1.left_stick_y); 
86           right = scale(gamepad1.left_stick_x); 
87           spin = scale(gamepad1.right_stick_x); 
88    
89           robot.move(forward, right, spin); 
90    
91           if (gamepad1.x || gamepad2.x) { 
92               robot.pushButton(-1); 
93           } 
94    
95    
96           if (gamepad1.b || gamepad2.b) { 
97               robot.pushButton(1); 
98           } 
99    
100          //reset servo if neither button is pressed 
101          if (!gamepad1.x && !gamepad2.x && !gamepad1.a && !gamepad2.a && !gamepad1.b && !gamepad2.b && !gamepad1.y && !gamepad2.y) { 
102              robot.pushButton(0); 
103          } 
104   
105          if (gamepad1.a || gamepad2.a) { 
106              robot.pushButton(robot.detectColor()); 
107          } 
108   
109          if (gamepad1.y || gamepad2.y) { 
110              robot.pushButton(robot.detectColor() * -1); 
111          } 
112   
113          if (gamepad1.left_bumper || gamepad2.left_bumper) { 
114              robot.spinner.setPower(1); 
115          } 
116   
117          if (gamepad1.right_bumper || gamepad2.right_bumper) { 
118              robot.spinner.setPower(-1); 
119          } 
120   
121          if (gamepad1.right_trigger > 0.1 || gamepad2.right_trigger > 0.1 || gamepad1.left_trigger > 0.1 || gamepad2.left_trigger > 0.1) { 
122              robot.spinner.setPower(0); 
123          } 
124   
125          // Send telemetry message to signify robot running; 
126          telemetry.addData("forward", "%.2f", forward); 
127          telemetry.addData("right", "%.2f", right); 
128          telemetry.addData("spin", "%.2f", spin); 
129      } 
130   
131      /* 
132       * Code to run ONCE after the driver hits STOP 
133       */ 
134      @Override 
135      public void stop() { 
136          robot.move(0, 0, 0); 
137      } 
138   
139      public double scale(double input) {//returns an interesting set of arcs 
140          if (input > 0) 
141              return (1 - Math.sqrt(1 - input * input)); 
142          else 
143              return (Math.sqrt(1 - input * input) - 1); 
144      } 
145   
146  }
