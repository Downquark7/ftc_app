25   package org.firstinspires.ftc.teamcode; 
26    
27   import com.qualcomm.robotcore.eventloop.opmode.OpMode; 
28   import com.qualcomm.robotcore.eventloop.opmode.TeleOp; 
29    
30   /** 
31    * Created by Chris D on 10/5/2016 
32    * <p> 
33    * In this example, you need to create a device configuration that lists two 
34    * "I2C Device"s, one named "mux" and the other named "ada". There are two 
35    * Adafruit color sensors plugged into the I2C multiplexer on ports 2 and 5. 
36    */ 
37   @TeleOp(name = "MultiplexColorSensorTest", group = "Iterative Opmode") 
38   //@Disabled 
39   public class MultiplexColorSensorTest extends OpMode { 
40       MultiplexColorSensor muxColor; 
41       int[] ports = {2, 5}; 
42    
43       @Override 
44       public void init() { 
45           int milliSeconds = 48; 
46           muxColor = new MultiplexColorSensor(hardwareMap, "mux", "ada", 
47                   ports, milliSeconds, 
48                   MultiplexColorSensor.GAIN_16X); 
49       } 
50    
51       @Override 
52       public void init_loop() { 
53    
54       } 
55    
56       @Override 
57       public void start() { 
58           muxColor.startPolling(); 
59       } 
60    
61       @Override 
62       public void loop() { 
63           for (int i = 0; i < ports.length; i++) { 
64               int[] crgb = muxColor.getCRGB(ports[i]); 
65    
66               telemetry.addLine("Sensor " + ports[i]); 
67               telemetry.addData("CRGB", "%5d %5d %5d %5d", 
68                       crgb[0], crgb[1], crgb[2], crgb[3]); 
69           } 
70    
71           telemetry.update(); 
72       } 
73    
74       @Override 
75       public void stop() { 
76    
77       } 
78   }
