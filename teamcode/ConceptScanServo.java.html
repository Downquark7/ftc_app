1    package org.firstinspires.ftc.teamcode; 
2     
3    import com.qualcomm.robotcore.eventloop.opmode.LinearOpMode; 
4    import com.qualcomm.robotcore.eventloop.opmode.TeleOp; 
5    import com.qualcomm.robotcore.hardware.Servo; 
6     
7    /** 
8     * This OpMode scans a single servo back and forwards until Stop is pressed. 
9     * The code is structured as a LinearOpMode 
10    * INCREMENT sets how much to increase/decrease the servo position each cycle 
11    * CYCLE_MS sets the update period. 
12    * <p> 
13    * This code assumes a Servo configured with the name "left claw" as is found on a pushbot. 
14    * <p> 
15    * NOTE: When any servo position is set, ALL attached servos are activated, so ensure that any other 
16    * connected servos are able to move freely before running this test. 
17    * <p> 
18    * Use Android Studio to Copy this Class, and Paste it into your team's code folder with a new name. 
19    * Remove or comment out the @Disabled line to add this opmode to the Driver Station OpMode list 
20    */ 
21   @TeleOp(name = "Concept: Scan Servo", group = "Concept") 
22   //@Disabled 
23   public class ConceptScanServo extends LinearOpMode { 
24    
25       static final double INCREMENT = 0.01;     // amount to slew servo each CYCLE_MS cycle 
26       static final int CYCLE_MS = 50;     // period of each cycle 
27       static final double MAX_POS = 1.0;     // Maximum rotational position 
28       static final double MIN_POS = 0.0;     // Minimum rotational position 
29    
30       // Define class members 
31       Servo servo; 
32       double position = (MAX_POS - MIN_POS) / 2; // Start at halfway position 
33       boolean rampUp = true; 
34    
35    
36       @Override 
37       public void runOpMode() { 
38    
39           // Connect to servo (Assume PushBot Left Hand) 
40           // Change the text in quotes to match any servo name on your robot. 
41           servo = hardwareMap.servo.get("buttonPusher"); 
42    
43           // Wait for the start button 
44           telemetry.addData(">", "Press Start to scan Servo."); 
45           telemetry.update(); 
46           waitForStart(); 
47    
48    
49           // Scan servo till stop pressed. 
50           while (opModeIsActive()) { 
51    
52               // slew the servo, according to the rampUp (direction) variable. 
53               if (gamepad1.dpad_up) 
54                   position += INCREMENT; 
55    
56               if (gamepad1.dpad_down) 
57                   position -= INCREMENT; 
58    
59               // Display the current value 
60               telemetry.addData("Servo Position", "%5.2f", position); 
61               telemetry.addData(">", "Press Stop to end test."); 
62               telemetry.update(); 
63    
64               // Set the servo to the new position and pause; 
65               servo.setPosition(position); 
66               sleep(CYCLE_MS); 
67               idle(); 
68           } 
69    
70           // Signal done; 
71           telemetry.addData(">", "Done"); 
72           telemetry.update(); 
73       } 
74   }
