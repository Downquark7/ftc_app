1    package org.firstinspires.ftc.teamcode; 
2     
3    import com.qualcomm.robotcore.eventloop.opmode.OpMode; 
4    import com.qualcomm.robotcore.eventloop.opmode.TeleOp; 
5    import com.qualcomm.robotcore.util.Range; 
6     
7    import static org.firstinspires.ftc.teamcode.robotconfig.dl; 
8     
9    /** 
10    * the main TeleOp program for our robot 
11    */ 
12    
13   @TeleOp(name = "TeleOp_Revised", group = "2016") 
14    
15   public class TeleOp_Revised extends OpMode { 
16    
17       private static double buttonPusher_MIN_RANGE = 0.35; 
18       private static double buttonPusher_MAX_RANGE = 0.75; 
19    
20       private static double Tilt_MAX_RANGE = 1.00; 
21       private static double Tilt_MIN_RANGE = 0.00; 
22    
23       private static double capRight_MAX_RANGE = 1.00; 
24       private static double capRight_MIN_RANGE = 0.00; 
25    
26       private static double capLeft_MAX_RANGE = 1.00; 
27       private static double capLeft_MIN_RANGE = 0.00; 
28       private static double buttonPusherDelta = 0.02; 
29       private static double tiltDelta = 0.02; 
30       private static double capLeftDelta = 0.02; 
31       private static double capRightDelta = 0.02; 
32       private static int pulses = 2240; 
33       robotconfig robot = new robotconfig(); 
34       private double buttonPusherPosition = 0; 
35       private double tiltPosition = 0; 
36       private double capLeftPosition = 0; 
37       private double capRightPosition = 0; 
38       private double forward = 0; 
39       private double right = 0; 
40       private double spin = 0; 
41       private double spinner = 0; 
42       private double reeler = 0; 
43       private double vexes; 
44       private double puncher; 
45       private boolean limitPuncher; 
46       private int endpulses = 0; 
47    
48       private boolean previousAState = false; 
49       private boolean previousGaryState = false; 
50       private boolean spinnerState = false; 
51       private boolean puncherState = false; 
52       private boolean capGrabberServoStateToggleFlag = true; 
53       private boolean capGrabberServoState = true; 
54       private boolean speedToggleFlag = false; 
55       private boolean slowState = false; 
56    
57       @Override 
58       public void init() { 
59    
60           robot.init(this); 
61           robot.move(0, 0, 0); 
62           robot.disableMotorEncoders(); 
63           // Send telemetry message to signify robot waiting; 
64           telemetry.addData("Say", "Hello Driver"); 
65           updateTelemetry(telemetry); 
66    
67           buttonPusherPosition = 0.5; 
68           tiltPosition = 0.50; 
69           capLeftPosition = 0.00; 
70           capRightPosition = 0.00; 
71    
72       } 
73    
74       @Override 
75       public void loop() { 
76    
77           forward = -gamepad1.left_stick_y * Math.abs(gamepad1.left_stick_y); 
78           right = gamepad1.left_stick_x * Math.abs(gamepad1.left_stick_x); 
79           spin = gamepad1.right_stick_x * Math.abs(gamepad1.right_stick_x); 
80           reeler = -gamepad2.right_stick_y; 
81    
82           if (slowState) { 
83               forward = forward * 0.25; 
84               right = right * 0.60; 
85               spin = spin * 0.35; 
86           } 
87    
88           /* 
89            *  This added code is to remove power from the cap ball grabber servos 
90            *  to be used if we need to disengage the arms and allow the to move 
91            *  freely. 
92            */ 
93    
94           robotconfig.addlog(dl, "in TeleOp_Revised", "top of main loop"); 
95    
96           if (gamepad1.right_bumper) {          // bumper is down 
97               if (speedToggleFlag) {   // was down last time, so ignore 
98                   // this time 
99               } else {                                // ok, this the first time 
100                  // through on a new buper down 
101                  speedToggleFlag = true; 
102   
103                  slowState = !slowState; 
104   
105              } 
106   
107          } else { 
108              speedToggleFlag = false;     // trigger is not pulled, so reset 
109              // trigger seen flag 
110          } 
111   
112   
113  //        if (gamepad1.right_bumper) { 
114  //            forward *= -1; 
115  //            right *= -1; 
116  //        } 
117   
118   
119          robot.move(forward, right, spin); 
120   
121          robot.reeler.setPower(reeler); 
122   
123          if (gamepad2.right_bumper) { 
124              vexes = -gamepad2.left_stick_y * 0.5 + 0.5; 
125              robot.rvex.setPosition(vexes); 
126              robot.lvex.setPosition(vexes); 
127          } 
128   
129          if (gamepad2.left_bumper) { 
130              capLeftPosition = 1; 
131              robot.capLeft.setPosition(capLeftPosition); 
132              capRightPosition = 1; 
133              robot.capRight.setPosition(capRightPosition); 
134              tiltPosition = 1; 
135              robot.tilt.setPosition(tiltPosition); 
136          } 
137   
138          if (gamepad1.y) { 
139              robot.puncher.setPower(1); 
140              puncherState = true; 
141          } 
142   
143          if (puncherState) { 
144              if (!robot.garry.isPressed() && previousGaryState) { 
145                  robot.puncher.setPower(0); 
146                  puncherState = false; 
147              } 
148          } 
149   
150          previousGaryState = robot.garry.isPressed(); 
151   
152          if (gamepad1.dpad_left) { 
153              buttonPusherPosition -= buttonPusherDelta; 
154              buttonPusherPosition = Range.clip(buttonPusherPosition, buttonPusher_MIN_RANGE, buttonPusher_MAX_RANGE); 
155              robot.buttonPusher.setPosition(buttonPusherPosition); 
156          } else if (gamepad1.dpad_right) { 
157              buttonPusherPosition += buttonPusherDelta; 
158              buttonPusherPosition = Range.clip(buttonPusherPosition, buttonPusher_MIN_RANGE, buttonPusher_MAX_RANGE); 
159              robot.buttonPusher.setPosition(buttonPusherPosition); 
160          } 
161   
162          if (gamepad1.left_bumper) { 
163              if (!previousAState) { 
164                  if (spinnerState) { 
165                      spinnerState = false; 
166                      robot.spinner.setPower(0); 
167                  } else { 
168                      spinnerState = true; 
169                      robot.spinner.setPower(1); 
170                  } 
171                  previousAState = true; 
172              } 
173          } else { 
174              previousAState = false; 
175          } 
176   
177          if (gamepad2.dpad_right) { 
178              capLeftPosition += capLeftDelta; 
179              capLeftPosition = Range.clip(capLeftPosition, capLeft_MIN_RANGE, capLeft_MAX_RANGE); 
180              robot.capLeft.setPosition(capLeftPosition); 
181          } else if (gamepad2.dpad_left) { 
182              capLeftPosition -= capLeftDelta; 
183              capLeftPosition = Range.clip(capLeftPosition, capLeft_MIN_RANGE, capLeft_MAX_RANGE); 
184              robot.capLeft.setPosition(capLeftPosition); 
185          } 
186   
187          if (gamepad2.x) { 
188              capRightPosition += capRightDelta; 
189              capRightPosition = Range.clip(capRightPosition, capRight_MIN_RANGE, capRight_MAX_RANGE); 
190              robot.capRight.setPosition(capRightPosition); 
191          } else if (gamepad2.b) { 
192              capRightPosition -= capRightDelta; 
193              capRightPosition = Range.clip(capRightPosition, capRight_MIN_RANGE, capRight_MAX_RANGE); 
194              robot.capRight.setPosition(capRightPosition); 
195          } 
196   
197          if (gamepad2.a) { 
198              tiltPosition -= tiltDelta; 
199              tiltPosition = Range.clip(tiltPosition, Tilt_MIN_RANGE, Tilt_MAX_RANGE); 
200              robot.tilt.setPosition(tiltPosition); 
201          } else if (gamepad2.y) { 
202              tiltPosition += tiltDelta; 
203              tiltPosition = Range.clip(tiltPosition, Tilt_MIN_RANGE, Tilt_MAX_RANGE); 
204              robot.tilt.setPosition(tiltPosition); 
205          } 
206   
207          if (gamepad2.back) 
208              robot.capLeft.getController().pwmDisable(); 
209          else if (gamepad2.left_stick_button) 
210              robot.capLeft.getController().pwmEnable(); 
211   
212          telemetry.addData("Forward", "%.2f", forward); 
213          telemetry.addData("Right", "%.2f", right); 
214          telemetry.addData("Spin", "%.2f", spin); 
215          telemetry.addData("ButtonPusher", "%.2f", buttonPusherPosition); 
216          telemetry.addData("Tilt", "%.2f", tiltPosition); 
217          telemetry.addData("capLeft", "%.2f", capLeftPosition); 
218          telemetry.addData("vexes", "%.2f", vexes); 
219          telemetry.addData("puncher", "%.2f", puncher); 
220          telemetry.addData("garry", "%b", !robot.garry.isPressed()); 
221   
222      } 
223   
224  }
