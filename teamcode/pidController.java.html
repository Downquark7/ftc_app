9    /*** 
10    * a class designed for making a simple pid controller... 
11    * It could probably be used with the gyroscope sensor to reduce rotation 
12    */ 
13   public class pidController { 
14       public double sp = 0;//set target for pv 
15    
16       public double pGain; 
17       public double iGain; 
18       public double dGain; 
19    
20       public double p = 0;//value to be calculated from the pid controller, public for tuning 
21       public double i = 0;//value to be calculated from the pid controller, public for tuning 
22       public double d = 0;//value to be calculated from the pid controller, public for tuning 
23    
24       public double ppv = 0;//previous process variable 
25       public double lowerIntegral = -1;//sets limit of intergral variable 
26       public double upperIntegral = 1;//sets limit of intergral variable 
27       public double e = 0;//error variable 
28    
29       public ElapsedTime time = new ElapsedTime(); 
30    
31       /*** 
32        * pidController creates a new PID controller. Could potentially be used for the Gyroscope sensor. 
33        * 
34        * @param pGain proportional gain constant 
35        * @param iGain integral gain constant 
36        * @param dGain derivative gain constant 
37        */ 
38       public pidController(double pGain, double iGain, double dGain) { 
39           this.pGain = pGain; 
40           this.iGain = iGain; 
41           this.dGain = dGain; 
42       } 
43    
44       public void setSp(double sp) { 
45           this.sp = sp; 
46       } 
47    
48       public void setLowerIntegral(double lowerIntegral) { 
49           this.lowerIntegral = lowerIntegral; 
50       } 
51    
52       public void setUpperIntegral(double upperIntegral) { 
53           this.upperIntegral = upperIntegral; 
54       } 
55    
56       public double getPID(double pv) { 
57           if (this.ppv == 0) 
58               this.ppv = pv; 
59    
60           this.e = (this.sp - pv); 
61           this.p = this.pGain * e; 
62           this.i = this.iGain * this.p; 
63    
64           if (this.i > this.upperIntegral) 
65               this.i = this.upperIntegral; 
66    
67           if (this.i < this.lowerIntegral) 
68               this.i = this.lowerIntegral; 
69    
70           this.d = -this.dGain * this.ppv * this.time.seconds(); 
71           this.ppv = pv; 
72           this.time.reset(); 
73           return (this.p + this.i + this.d); 
74       } 
75   }
