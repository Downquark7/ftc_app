49           robot.init(this);

61       @Override

63    
64           forward = -gamepad1.left_stick_y; 
65           right = gamepad1.left_stick_x; 
66           spin = gamepad1.right_stick_x; 
67    
68           if (gamepad1.right_bumper || gamepad1.left_bumper) { 
69               forward *= -1; 
70               right *= -1; 
71           } 
72    
73           forward = Range.clip(forward, -1, 1); 
74           right = Range.clip(right, -1, 1); 
75           spin = Range.clip(spin, -1, 1); 
76           spinner = Range.clip(spinner, -1, 1); 
77           reeler = Range.clip(reeler, -1, 1); 
78           vexes = Range.clip(vexes, -1, 1); 
79           cam = Range.clip(cam, -1, 1); 
80    
81           robot.move(forward, right, spin); 
82    
83           robot.spinner.setPower(spinner); 
84           robot.reeler.setPower(reeler); 
85    
86    
87           if (gamepad1.dpad_left || gamepad1.x) { 
88               robot.pushSoftButton(-1); 
89           } 
90    
91           if (gamepad1.dpad_right || gamepad1.b) { 
92               robot.pushSoftButton(1); 
93           } 
94    
95           if (gamepad2.a) { 
96               spinner = -gamepad2.left_stick_y; 
97           } 
98    
99           if (gamepad2.b) { 
100              vexes = -gamepad2.left_stick_y * 0.5 + 0.5; 
101                  robot.lvex.setPosition(vexes); 
102                  robot.rvex.setPosition(vexes); 
103          } 
104   
105          if (gamepad2.right_bumper) { 
106              try { 
107                  robot.puncher.setPower(1); 
108              } catch (Exception err) { 
109                  robotconfig.addlog(dl, "error", "failed to set power of puncher"); 
110              } 
111          } else { 
112              try { 
113                  robot.puncher.setPower(0); 
114              } catch (Exception err) { 
115                  robotconfig.addlog(dl, "error", "failed to set power of puncher"); 
116              } 
117          } 
118   
119          reeler = -gamepad2.right_stick_y; 
120   
121          if (gamepad2.dpad_up) { 
122              capLeftPosition = 0.66; 
123              tiltPosition = 1.00; 
124          } 
125   
126          if (gamepad2.dpad_right) { 
127              capLeftPosition = 0.90; 
128              tiltPosition = 1.00; 
129          } 
130   
131          if (gamepad2.dpad_down) { 
132              capLeftPosition = 0.90; 
133              tiltPosition = 0.00; 
134          }

137              capLeftPosition = 0.66; 
138              tiltPosition = 0.00;

142              capLeftPosition = 0; 
143              tiltPosition = 0.50; 
144          }

146          tiltPosition = Range.clip(tiltPosition, Tilt_MIN_RANGE, Tilt_MAX_RANGE); 
147          capRightPosition = Range.clip(capRightPosition, capRight_MIN_RANGE, capRight_MAX_RANGE); 
148          capLeftPosition = Range.clip(capLeftPosition, capLeft_MIN_RANGE, capLeft_MAX_RANGE);

150          robot.tilt.setPosition(tiltPosition);

152          robot.capRight.setPosition(capLeftPosition);

157          telemetry.addData("Tilt", "%.2f", tiltPosition);
