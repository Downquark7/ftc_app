45       /*** 
46        * uses math to convert an amount of degrees into how much distance the wheel spins 
47        * 
48        * @param degrees amount of degrees to spin clockwise, negative if backwards 
49        * @return distance a wheel will spin in mm 
50        */ 
51       public double spin2mm(double degrees) { 
52           return (degrees / 360) * (measurements.wheelDiagonal * measurements.pi); 
53       } 
54    
55       /*** 
56        * uses math to convert a distance in millimeters to the number of pulses the motor should generate to go that far 
57        * 
58        * @param mm a distance in millimeters 
59        * @return number of pulses generated 
60        */ 
61       public int mm2pulses(double mm) { 
62           return (int) ((measurements.ppr / (measurements.pi * measurements.wheelDiameter)) * mm);//moved sprocket change to measurements.ppr 
63       }

65       /*** 
66        * function takes measurements in mm to move the robot 
67        * 
68        * @param forward   amount to move forward in mm 
69        * @param right     amount to move right in mm

71        * @param timeout   time to take doing movement 
72        * @param robot     just make it robot, for getting the configuration 
73        * @param telemetry just leave it as telemetry to allow program to write telemetry

75       public void move(double forward, double right, double spin, double timeout, robotconfig robot, Telemetry telemetry) { 
76           robotconfig.addlog(dl, "pm.move with telemetry", "called with right:" + String.format(Locale.ENGLISH, "%.2f", right) + " and spin:" + String.format(Locale.ENGLISH, "%.2f", spin)); 
77           telemetry.addData("ods", "%d", robot.ods.getLightDetected()); 
78           robotconfig.addlog(dl, "pm.move", "called with right:" + String.format(Locale.ENGLISH, "%.2f", right) + " and spin:" + String.format(Locale.ENGLISH, "%.2f", spin)); 
79           robot.setMotorTargets(mm2pulses(forward), mm2pulses(right), mm2pulses(spin2mm(spin))); 
80           waitForMotors(robot, telemetry, forward, right, spin, timeout); 
81       }

98       public void movesegments(double forward, double right, double spin, double timeout, int segments, robotconfig robot, Telemetry telemetry) { 
99           for (int i = 0; i < segments; i++) { 
100              move(forward / segments, right / segments, spin / segments, timeout / segments, robot, telemetry); 
101          }

104      private void waitForMotors(robotconfig robot, Telemetry telemetry, double forward, double right, double spin, double timeout) { 
105          robotconfig.addlog(dl, "pm.waitforMotors with telemetry", "called with right:" + String.format(Locale.ENGLISH, "%.2f", right) + " and spin:" + String.format(Locale.ENGLISH, "%.2f", spin) + " and forward:" + String.format(Locale.ENGLISH, "%.2f", forward)); 
106          runtime.reset(); 
107          while (robot.isMotorBusy() && (runtime.seconds() < timeout) && robotconfig.theLinearOpMode.opModeIsActive()) { 
108              telemetry.addData("Path0", "Go to %f in : %f in : %f in", (forward / measurements.mmPerInch), (right / measurements.mmPerInch), spin); 
109              telemetry.addData("Path0", "Go to %f in : %f in : %f in", (forward / measurements.mmPerInch), (right / measurements.mmPerInch), spin); 
110              telemetry.addData("Path1", "At  %7d :%7d :%7d :%7d", robot.fLeftMotor.getCurrentPosition(), robot.fRightMotor.getCurrentPosition(), robot.bLeftMotor.getCurrentPosition(), robot.bRightMotor.getCurrentPosition()); 
111              telemetry.addData("Path2", "End %7d :%7d :%7d :%7d", robot.fLeftMotor.getTargetPosition(), robot.fRightMotor.getTargetPosition(), robot.bLeftMotor.getTargetPosition(), robot.bRightMotor.getTargetPosition()); 
112              telemetry.update(); 
113              Thread.yield(); 
114          } 
115          // Need to gracefully exit loop here as we have either timed out or a stop has been requested 
116   
117          if (runtime.seconds() >= timeout) { 
118              robotconfig.addlog(dl, "pm.waitforMotors", "timed out: " + String.format(Locale.ENGLISH, "%.2f", runtime.seconds())); 
119          } else if (!robotconfig.theLinearOpMode.opModeIsActive()) { 
120              robot.resetMotorEncoders(); 
121              robotconfig.addlog(dl, "pm.waitforMotors", "Stop of opmode was requested"); 
122          } else { 
123              robotconfig.addlog(dl, "pm.waitforMotors", "exited move normally"); 
124          } 
125      }
